function enviarInscripcion() { // Recoger datos del formulario const name = document.getElementById("inscName").value; const age = parseInt(document.getElementById("inscAge").value) || 0; const phone = document.getElementById("inscPhone").value; const email = document.getElementById("inscEmail").value; const modalidad = document.getElementById("inscModalidad").value; let msg = ""; if (modalidad === "individual") { // Preparar inscripción individual msg = `Inscripción recibida: ${name}, ${age} años\nTel: ${phone} | Email: ${email}\nModalidad: Individual`; // Crear objeto para inscripción individual const athlete = { type: "individual", bib: "", // Puedes generar un número o dejarlo vacío name: name, age: age, gender: "", // Si deseas, puedes agregar un selector de género en el formulario rango: calcularRangoEdad(age), time: "", heat: "", public: false, paid: false }; // Guardar en la base de datos persistente athletesDB.push(athlete); saveAthletesDB(athletesDB); refreshAthletesTable(); } else { // Inscripción en pareja const name2 = document.getElementById("inscName2").value; const age2 = parseInt(document.getElementById("inscAge2").value) || 0; msg = `Inscripción recibida:\n- ${name}, ${age} años\n- ${name2}, ${age2} años\nTel: ${phone} | Email: ${email}\nModalidad: Pareja`; // Crear objeto para inscripción en pareja const athletePair = { type: "pair", bib: "", name1: name, age1: age, rango1: calcularRangoEdad(age), name2: name2, age2: age2, rango2: calcularRangoEdad(age2), pairCat: "pareja", // Puedes definir la categoría según tus criterios time: "", heat: "", public: false, paid: false }; // Guardar en la base de datos persistente athletesDB.push(athletePair); saveAthletesDB(athletesDB); refreshAthletesTable(); } // Mostrar el mensaje de confirmación document.getElementById("inscResultMsg").textContent = msg; }
